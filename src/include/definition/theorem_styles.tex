% ~ Theorem Style Definitions
\newtheoremstyle{def_style}%                % Name
{10pt}%                                     % Space above
{10pt}%                                     % Space below
% {\itshape}%
{}                             % Body font
{0pt}%                                     % Indent amount
{\bfseries}%                            % Theorem head font
{ }%                                    % Punctuation after theorem head
{.5em}%                                    % Space after theorem head, ' ', or \newline
{}% 

\newtheoremstyle{thm_style}%                % Name
{15pt}%                                     % Space above
{15pt}%                                     % Space below
% {\itshape}%                             % Body font
{}
{0pt}%                                     % Indent amount
{\bfseries}%                            % Theorem head font
{ }%                                    % Punctuation after theorem head
{\newline}%                                    % Space after theorem head, ' ', or \newline
{}% 

\newtheoremstyle{hyp_style} % name
{3pt}% Space above
{3pt}% Space below
{\itshape}% Body font
{}% Indent amount
{\bfseries}% Theorem head font
{}% Punctuation after theorem head
{0pt}% Space after theorem head
{(S\thmnumber{#2})}% Theorem head spec (can be left empty, meaning ‘normal’)

\newtheoremstyle{rem_style}%                % Name
{5pt}%                                     % Space above
{5pt}%                                     % Space below
% {\itshape}%
{}                             % Body font
{0pt}%                                     % Indent amount
{\emph}%                            % Theorem head font
{ :}%                                    % Punctuation after theorem head
{.5em}%                                    % Space after theorem head, ' ', or \newline
{}%
