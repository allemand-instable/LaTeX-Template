% enumitem

% & Name
% 		checkmarks
% & Description
% 		enum environment where items have a box as bullets □, you can add to these boxes either ✓ or × to the checkbox with \checked and \crossed
\newlist{checkmarks}{itemize}{2}
\setlist[checkmarks]{label=$\square$}

% & Name
% 		circledenum
% & Description
% 		Enum environement where items are labeled with numbers that are circled : (𝟏), (𝟐), ...
\newlist{circledenum}{enumerate}{1}
\setlist[circledenum]{label=\protect\circled{\arabic*}}

% pifont
\newcommand{\cmark}{\ding{51}}%
\newcommand{\xmark}{\ding{55}}%

% & Name
% 		checked / crossed
% & Description
% 		either add ✓ or × to the checkbox
% & Environment
% 		checkmarks
% & Arguments
%		None
% & Examples
% 		\item[\checked] sample text
%					▶ [✓]
%		\item[\crossed] sample text
%					▶ [×]
\newcommand{\checked}{\rlap{$\square$}{\raisebox{2pt}{\large\hspace{1pt}\cmark}}%
	\hspace{-2.5pt}}
\newcommand{\crossed}{\rlap{$\square$}{\large\hspace{1pt}\xmark}}
