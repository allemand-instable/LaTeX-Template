% & Name

% 		Spectre d'un endomorphisme ou opÃ©rateur T : sp(T)

% 		optionnel [] : corps du spectre

% 		\Sp{u} : sp(u)

\NewDocumentCommand{\Sp}{s o m}{%

    % Starred version: \sp*

      % \Sp[K]{u} -> sp_K(u)

      % \Sp{u} -> sp(u)

    % Non-starred version: \sp
    \IfValueTF{#2}
    {%
      % \Sp[K]{u} -> sp_K(u) 
      \operatorname{sp}_{#2}\bigl({#3}\bigr)%
    }
    {%
      % \Sp{u} -> sp(u)
      \operatorname{sp}\bigl({#3}\bigr)%
    }%
  }%
}

\NewDocumentCommand{\inlinefunc}{s m o m o}{

          % !     *{x}[E]{y}

        }%

          % !     *{x}{y}[F]
          {#2} \mapsto {#4} \in {#5}%
        }{%
          % !     *{x}{y}
          {#2} \mapsto {#4}%
        }%
      }%

      % ~ non star

      \IfValueTF{#3}{%
        \IfValueTF{#5}{%
          % !   {E}[x]{F}[y]
          {#3} \rightarrow {#5} \,:\, {#2} \mapsto {#4}%
        }{%
          % !   {E}[x]{F}
          {#3} \in {#2} \rightarrow {#4}%
        }%
      }{%

          % !   {E}{F}[y]
          {#2} \rightarrow {#4} \ni {#5}%
        }{%
          % !   {E}{F}
          {#2} \rightarrow {#4}%
        }%
      }%
    }%
