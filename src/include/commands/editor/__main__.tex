% & Name
%       \citationrequise
% & Description
%       Indique dans le rendu du PDF en couleur orange qu'une citation est requise à cet endroit
% & Arguments
%       None
% & Examples
%       On sait que les fonctions dérivables nulle part sont denses pour la norme de la convergence uniforme dans les fonctions continues \citationrequise
\newcommand{\citationrequise}{\textcolor{flatuicolors_orange}{\faEdit ( \faExclamationTriangle citation requise \faFile*[regular] )}}

% & Name
%       \exemplerequis
% & Description
%       add a colored warning that is rendered in the output PDF to mention that examples are needed
% & Arguments
%       None
% & Examples
%       y a plein de fonctions $f$ non continues \exemplerequis
\newcommand{\exemplerequis}{\textcolor{flatuicolors_orange}{\faEdit ( exemple concret requis )}}

% & Name
%       \editorwarn
% & Description
%       add a colored warning that is rendered in the output PDF
% & Arguments
%       1: message
% & Examples
%       \editorwarn{attention, toutes les fonctions continues ne sont pas holderiennes}
\newcommand{\editorwarn}[1]{\textcolor{flatuicolors_orange}{\faExclamationTriangle \, (  #1  )}}

% & Name
%       \editlater
% & Description
%       add a colored warning that is rendered in the output PDF that mentions that further editing will be required here
% & Arguments
%       None
% & Examples
%       Ce texte est pas très bien rédigé \editlater
\newcommand{\editlater}[1]{\textcolor{flatuicolors_blue_deep}{\faEdit \, (  #1  )}}
